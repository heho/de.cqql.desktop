1 dojo.provide('de.cqql.desktop.Desktop');
  2 
  3 dojo.require('de.cqql.desktop.WindowManager');
  4 dojo.require('de.cqql.desktop.WindowBar');
  5 
  6 dojo.require('de.cqql.desktop.menu.AppMenu');
  7 
  8 dojo.declare('de.cqql.desktop.Desktop', null, {
  9 	/**
 10 	 * The desktop
 11 	 *
 12 	 * @lends de.cqql.desktop.Desktop
 13 	 */
 14 	
 15 	_appMenu: null,
 16 	
 17 	_windowManager: null,
 18 	
 19 	_rootNode: null,
 20 	
 21 	_windowBar: null,
 22 	
 23 	/**
 24 	 * @constructs
 25 	 */
 26 	constructor: function (rootNode)
 27 	{
 28 		this.setRootNode(rootNode);
 29 	},
 30 	
 31 	setRootNode: function (rootNode)
 32 	{
 33 		this._rootNode = rootNode;
 34 	},
 35 	
 36 	getRootNode: function ()
 37 	{
 38 		return this._rootNode;
 39 	},
 40 	
 41 	setAppMenu: function (appMenu)
 42 	{
 43 		this._appMenu = appMenu;
 44 	},
 45 	
 46 	getAppMenu: function ()
 47 	{
 48 		if (this._appMenu == null)
 49 		{
 50 			this.setAppMenu(new de.cqql.desktop.menu.AppMenu());
 51 		}
 52 		
 53 		return this._appMenu;
 54 	},
 55 	
 56 	setWindowManager: function (windowManager)
 57 	{
 58 		this._windowManager = windowManager;
 59 	},
 60 	
 61 	getWindowManager: function ()
 62 	{
 63 		if (this._windowManager == null)
 64 		{
 65 			this.setWindowManager(
 66 				new de.cqql.desktop.WindowManager(
 67 					this.getRootNode(),
 68 					this.getWindowBar()
 69 				)
 70 			);
 71 		}
 72 		
 73 		return this._windowManager;
 74 	},
 75 	
 76 	setWindowBar: function (windowBar)
 77 	{
 78 		this._windowBar = windowBar;
 79 	},
 80 	
 81 	getWindowBar: function ()
 82 	{
 83 		if (this._windowBar == null)
 84 		{
 85 			this.setWindowBar(new de.cqql.desktop.WindowBar());
 86 		}
 87 		
 88 		return this._windowBar;
 89 	},
 90 	
 91 	run: function ()
 92 	{
 93 		this.getWindowManager();
 94 	}
 95 });
